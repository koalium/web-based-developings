Evaporative Ventilation Control Algorithm

Air Flow Calculation
// Return air from saloon
RT = ReadTemperature(SENSOR_INDOOR_TEMP);
RH = ReadHumidity(SENSOR_INDOOR_HUM);

// Outdoor conditions
OT = ReadTemperature(SENSOR_OUTDOOR_TEMP);
OH = ReadHumidity(SENSOR_OUTDOOR_HUM);

// Calculate air percentages
ExhaustPercentage = Dampers.Exhaust_Position;
FreshPercentage = Dampers.Fresh_Position;
ReturnPercentage = 100 - (ExhaustPercentage + FreshPercentage);

// Mixing calculation
MixedAirTemp = (OT * FreshPercentage + RT * ReturnPercentage) / 100;
MixedAirHum = (OH * FreshPercentage + RH * ReturnPercentage) / 100;

// Heating cycle if needed
if (Heater.Valve_Open) {
    CycleTemp = MixedAirTemp + HEATING_COEFFICIENT;
} else {
    CycleTemp = MixedAirTemp;
}

// Evaporative cooling if pump is ON
if (Pump.Running) {
    FinalTemp = CalculateWetBulb(CycleTemp, MixedAirHum);
    FinalHum = Math.min(85, MixedAirHum + 20); // Max 85% humidity
} else {
    FinalTemp = CycleTemp;
    FinalHum = MixedAirHum;
}

// Control outputs
FanFeed.Speed = PID(FinalTemp, SETPOINT_TEMP);
FanReturn.Speed = FanFeed.Speed * 0.9;
