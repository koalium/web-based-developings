// Air Flow Calculation
FUNCTION CalculateAirFlow : REAL
VAR_INPUT
    SaloonTemp: REAL;    // ST - Saloon Temperature
    SaloonHumidity: REAL; // SH - Saloon Humidity
    OutdoorTemp: REAL;    // OT - Outdoor Temperature
    OutdoorHumidity: REAL; // OH - Outdoor Humidity
    ExhaustPercentage: REAL; // 0-100%
    HeaterValve: BOOL;
    PumpStatus: BOOL;
END_VAR
VAR_OUTPUT
    FeedTemp: REAL;
    FeedHumidity: REAL;
    CycleTemp: REAL;
END_VAR
VAR
    ReturnAir: REAL;
    ExhaustAir: REAL;
    RemainAir: REAL;
    CycleAir: REAL;
    MixedAir: REAL;
    FreshAir: REAL;
    ED_Efficiency: REAL := 1.0;
    HeatingConstant: REAL := 1.2;
    CoolingConstant: REAL := 0.85;
END_VAR

// Calculate return air from saloon
ReturnAir := SaloonTemp;

// Calculate exhaust air
ExhaustAir := ExhaustPercentage * ReturnAir * ED_Efficiency;

// Calculate remaining air
RemainAir := ReturnAir - ExhaustAir;

// Calculate cycle air (with optional heating)
IF HeaterValve THEN
    CycleAir := RemainAir + HeatingConstant;
ELSE
    CycleAir := RemainAir;
END_IF

// Calculate fresh air percentage
FreshAir := ExhaustPercentage;
CycleAirPercentage := 100 - FreshAir;

// Calculate mixed air (psychometric mixing)
MixedAir := (FreshAir * OutdoorTemp + CycleAirPercentage * CycleAir) / 100;

// Calculate feed air (with optional humidification)
IF PumpStatus THEN
    FeedTemp := MixedAir * CoolingConstant;
    FeedHumidity := MIN(85.0, OutdoorHumidity * 1.1);
ELSE
    FeedTemp := MixedAir;
    FeedHumidity := OutdoorHumidity;
END_IF

// Return results
FeedTemp := FeedTemp;
FeedHumidity := FeedHumidity;
CycleTemp := CycleAir;

END_FUNCTION
